```csharp
using System;
using System.Text.RegularExpressions;

namespace MyApp.Core.Helpers
{
    static class TextUtils
    {
        public static string Slugify(string input)
        {
            if (string.IsNullOrWhiteSpace(input))
                throw new ArgumentException("Input cannot be empty.", nameof(input));

            var lower = input.Trim().ToLowerInvariant();
            return Regex.Replace(lower, @"[^a-z0-9]+", "-").Trim('-');
        }

        public static string Truncate(string text, int maxLength, string suffix = "...")
        {
            if (text.Length <= maxLength)
                return text;

            return text[..(maxLength - suffix.Length)] + suffix;
        }

        private static bool IsAsciiLetter(char c) =>
            (c >= 'a' && c <= 'z') || (c >= 'A' && c <= 'Z');
    }
}
```

```csharp
using System;
using System.Collections.Generic;
using System.Linq;

namespace MyApp.Domain
{
    public class Order
    {
        private readonly List<OrderItem> _items = new();

        public Guid Id { get; } = Guid.NewGuid();
        public DateTime CreatedAt { get; } = DateTime.UtcNow;
        public OrderStatus Status { get; private set; } = OrderStatus.Pending;
        public decimal Total => _items.Sum(i => i.Subtotal);

        public void AddItem(string productId, int quantity, decimal unitPrice)
        {
            if (quantity <= 0)
                throw new ArgumentOutOfRangeException(nameof(quantity), "Quantity must be positive.");

            _items.Add(new OrderItem(productId, quantity, unitPrice));
        }

        public void Confirm()
        {
            if (Status != OrderStatus.Pending)
                throw new InvalidOperationException($"Cannot confirm order in {Status} state.");

            Status = OrderStatus.Confirmed;
        }

        public IReadOnlyList<OrderItem> GetItems() => _items.AsReadOnly();
    }
}
```

```csharp
using System;
using System.Collections.Generic;
using System.Linq;

namespace MyApp.Core
{
    class ReportBuilder
    {
        public IEnumerable<string> BuildSummary(IEnumerable<Order> orders)
        {
            var lines = new List<string>();
            var grouped = orders.GroupBy(o => o.Status);

            foreach (var group in grouped)
            {
                decimal total = group.Sum(o => o.Total);
                lines.Add($"{group.Key}: {group.Count()} orders — R${total:F2}");
            }

            for (int i = 0; i < lines.Count; i++)
                lines[i] = $"[{i + 1}] {lines[i]}";

            while (lines.Count > 0 && string.IsNullOrWhiteSpace(lines[^1]))
                lines.RemoveAt(lines.Count - 1);

            return lines;
        }
    }
}
```

### Error Handling

```csharp
using System;
using System.IO;

namespace MyApp.Infrastructure
{
    class FileImporter
    {
        private readonly string _basePath;

        public FileImporter(string basePath)
        {
            _basePath = basePath;
        }

        public string Import(string fileName)
        {
            var path = Path.Combine(_basePath, fileName);
            try
            {
                return File.ReadAllText(path);
            }
            catch (FileNotFoundException ex)
            {
                throw new ImportException($"File not found: {fileName}", ex);
            }
            catch (UnauthorizedAccessException ex)
            {
                throw new ImportException($"Permission denied: {fileName}", ex);
            }
            finally
            {
                Console.WriteLine($"[FileImporter] Attempted: {fileName}");
            }
        }
    }

    class ImportException : Exception
    {
        public ImportException(string message, Exception inner) : base(message, inner) { }
    }
}
```

### Async / Concurrency

```csharp
using System;
using System.Net.Http;
using System.Text.Json;
using System.Threading;
using System.Threading.Tasks;

namespace MyApp.Infrastructure
{
    class MarketDataClient
    {
        private readonly HttpClient _http;
        private readonly string _baseUrl;

        public MarketDataClient(HttpClient http, string baseUrl)
        {
            _http = http;
            _baseUrl = baseUrl;
        }

        public async Task<T> GetAsync<T>(string endpoint, CancellationToken ct = default)
        {
            var response = await _http.GetAsync($"{_baseUrl}/{endpoint}", ct);
            response.EnsureSuccessStatusCode();

            var json = await response.Content.ReadAsStringAsync(ct);
            return JsonSerializer.Deserialize<T>(json)
                ?? throw new InvalidOperationException("Empty response body.");
        }

        public async Task<string[]> GetTickersAsync(CancellationToken ct = default) =>
            await GetAsync<string[]>("tickers", ct);
    }
}
```

### Generics

```csharp
using System;
using System.Collections.Generic;

namespace MyApp.Data
{
    class Repository<TEntity, TKey>
        where TEntity : class
        where TKey : IEquatable<TKey>
    {
        private readonly Dictionary<TKey, TEntity> _store = new();

        public void Save(TKey id, TEntity entity)
        {
            ArgumentNullException.ThrowIfNull(entity);
            _store[id] = entity;
        }

        public bool TryGet(TKey id, out TEntity? entity) =>
            _store.TryGetValue(id, out entity);

        public void Delete(TKey id)
        {
            if (!_store.Remove(id))
                throw new KeyNotFoundException($"Entity with key '{id}' not found.");
        }

        public IReadOnlyDictionary<TKey, TEntity> All => _store;
    }
}
```
